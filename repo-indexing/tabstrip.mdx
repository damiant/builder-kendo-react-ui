# TabStrip Component

## Overview

### Purpose
Use this component to generate tab strip functionality

### When to Use
The TabStrip component is ideal for organizing content into multiple views within a single container, allowing users to switch between different sections seamlessly. Common use cases include:

- **Content organization** - Group related information into logical sections like Profile, Settings, and Security
- **Multi-step forms** - Break complex forms into manageable sections with tab-based navigation
- **Dashboard panels** - Organize different views of data like Charts, Tables, and Reports
- **Application sections** - Separate application features like Documents, Messages, and Calendar
- **Preference panels** - Organize user settings into categories like General, Privacy, and Notifications
- **Product information** - Display product details across tabs like Description, Specifications, and Reviews
- **Interface layout** - Provide horizontal navigation for content areas in limited screen space
- **Responsive design** - Offer space-efficient navigation on smaller screens and devices

### Architecture
**No Coupling**: TabStrip is a standalone component that operates independently without requiring other components. It provides a complete tabbed interface solution with built-in state management, keyboard navigation, accessibility features, and customizable positioning.

## Components

### TabStrip

**Purpose:** Provides a container for organizing multiple Tab components with built-in navigation, state management, and accessibility features for seamless content switching.

**Interface:**
```typescript
interface TabStripProps {
  // State management
  selected?: number; // Index of the currently selected tab (controlled component)
  onSelect?: (event: TabStripSelectEvent) => void; // Callback when tab selection changes
  
  // Visual configuration
  animation?: boolean; // Enable/disable animation transitions between tabs (default: true)
  tabPosition?: 'top' | 'bottom' | 'left' | 'right'; // Position of the tab navigation (default: 'top')
  
  // Content structure
  children?: React.ReactNode; // Tab components that define the tabbed content
  
  // Styling
  className?: string; // Custom CSS classes for the TabStrip container
  style?: React.CSSProperties; // Inline styles for the TabStrip container
  dir?: 'ltr' | 'rtl'; // Text direction for internationalization
  
  // Form integration
  name?: string; // Form field name for form submissions
  
  // Accessibility
  id?: string; // Unique identifier for the TabStrip
  tabIndex?: number; // Tab order position for keyboard navigation
}

interface TabStripSelectEvent {
  selected: number; // Index of the newly selected tab
  target: TabStrip; // Reference to the TabStrip component
}

interface TabProps {
  // Content definition
  title?: React.ReactNode; // Tab header text or React element
  children?: React.ReactNode; // Content displayed when tab is active
  
  // Interaction control
  disabled?: boolean; // Disable tab interaction and selection
  
  // Styling
  contentClassName?: string; // CSS class applied to the tab content area
}
```

### Dynamic Tab Management
```tsx
function DynamicTabManager() {
  const [tabs, setTabs] = useState([
    { id: 1, title: 'Home', content: 'Welcome to the home page', closable: false },
    { id: 2, title: 'Profile', content: 'User profile information', closable: true },
    { id: 3, title: 'Settings', content: 'Application settings', closable: true }
  ]);
  
  const [selectedTab, setSelectedTab] = useState(0);
  const [nextId, setNextId] = useState(4);

  const addNewTab = () => {
    const newTab = {
      id: nextId,
      title: `Tab ${nextId}`,
      content: `Content for tab ${nextId}`,
      closable: true
    };
    
    setTabs(prev => [...prev, newTab]);
    setSelectedTab(tabs.length); // Select the new tab
    setNextId(prev => prev + 1);
  };

  const closeTab = (indexToClose: number) => {
    if (tabs[indexToClose]?.closable) {
      setTabs(prev => prev.filter((_, index) => index !== indexToClose));
      
      // Adjust selected tab index if necessary
      if (indexToClose <= selectedTab) {
        setSelectedTab(Math.max(0, selectedTab - 1));
      }
    }
  };

  const handleTabSelect = (event: TabStripSelectEvent) => {
    setSelectedTab(event.selected);
  };

  return (
    <div className="dynamic-tab-manager">
      <div className="tab-controls">
        <Button onClick={addNewTab} icon="plus">
          Add New Tab
        </Button>
      </div>
      
      <TabStrip
        selected={selectedTab}
        onSelect={handleTabSelect}
        className="dynamic-tabs"
      >
        {tabs.map((tab, index) => (
          <Tab 
            key={tab.id}
            title={
              <div className="tab-header">
                <span>{tab.title}</span>
                {tab.closable && (
                  <button 
                    className="close-tab-btn"
                    onClick={(e) => {
                      e.stopPropagation();
                      closeTab(index);
                    }}
                    aria-label={`Close ${tab.title} tab`}
                  >
                    ×
                  </button>
                )}
              </div>
            }
          >
            <div className="tab-content">
              <h3>{tab.title}</h3>
              <p>{tab.content}</p>
              <div className="tab-info">
                <small>Tab ID: {tab.id}</small>
              </div>
            </div>
          </Tab>
        ))}
      </TabStrip>
    </div>
  );
}
```

### Responsive TabStrip
```tsx
function ResponsiveTabStrip() {
  const [tabPosition, setTabPosition] = useState<'top' | 'bottom' | 'left' | 'right'>('top');
  const [isMobile, setIsMobile] = useState(false);

  useEffect(() => {
    const handleResize = () => {
      const mobile = window.innerWidth < 768;
      setIsMobile(mobile);
      
      // Switch to bottom position on mobile for better accessibility
      if (mobile) {
        setTabPosition('bottom');
      } else {
        setTabPosition('top');
      }
    };

    window.addEventListener('resize', handleResize);
    handleResize(); // Check initial size
    
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  const [currentTab, setCurrentTab] = useState(0);

  return (
    <div className="responsive-container">
      {!isMobile && (
        <div className="position-controls">
          <h4>Tab Position:</h4>
          {(['top', 'bottom', 'left', 'right'] as const).map(position => (
            <label key={position}>
              <input
                type="radio"
                name="tabPosition"
                value={position}
                checked={tabPosition === position}
                onChange={(e) => setTabPosition(e.target.value as any)}
              />
              {position.charAt(0).toUpperCase() + position.slice(1)}
            </label>
          ))}
        </div>
      )}
      
      <TabStrip
        selected={currentTab}
        onSelect={(e) => setCurrentTab(e.selected)}
        tabPosition={tabPosition}
        animation={!isMobile} // Disable animation on mobile for performance
        className={`responsive-tabs ${isMobile ? 'mobile' : 'desktop'}`}
      >
        <Tab title="Home">
          <div className="responsive-content">
            <h2>Home</h2>
            <p>Welcome to our responsive application. This content adapts to different screen sizes.</p>
            {isMobile && <p><strong>Mobile view detected</strong></p>}
```

## Best Practices

### Usage Guidelines
- **Organize content logically** - Group related information into tabs that follow a natural progression or categorization
- **Use descriptive tab titles** - Provide clear, concise titles that accurately describe the content within each tab
- **Limit tab count** - Keep the number of tabs manageable (typically 5-7 tabs maximum) to avoid overwhelming users
- **Consider tab order** - Place the most important or frequently accessed content in the first tabs
- **Maintain consistent content structure** - Use similar layouts and patterns across different tabs for a cohesive experience

### Accessibility
- **Keyboard navigation** - TabStrip supports full keyboard accessibility with arrow keys for tab navigation and Tab key for content focus
- **Screen reader support** - The component provides proper ARIA roles and announces tab states and content to assistive technologies
- **Focus management** - Clear visual focus indicators help users navigate between tabs and tab content effectively
- **Semantic structure** - Use proper heading hierarchies and semantic HTML within tab content for better screen reader navigation
- **Tab state communication** - Disabled tabs are properly announced to screen readers with appropriate ARIA attributes

### Performance
- **Content lazy loading** - Consider lazy loading tab content that isn't immediately visible to improve initial page load performance
- **Animation considerations** - Disable animations on mobile devices or slower connections to improve perceived performance
- **State management** - Use controlled components for complex state scenarios, uncontrolled for simple implementations
- **Event handler optimization** - Use useCallback hooks to prevent unnecessary re-renders when tab content is complex

```tsx
function ApplicationConfig() {
  const [config, setConfig] = useState({
    general: {
      appName: 'My Application',
      language: 'en-US',
      timezone: 'UTC-8',
      theme: 'light'
    },
    database: {
      host: 'localhost',
      port: 5432,
      name: 'myapp_db',
      ssl: true
    },
    api: {
      baseUrl: 'https://api.example.com',
      timeout: 30000,
      retries: 3,
      apiKey: '***hidden***'
    },
    security: {
      sessionTimeout: 3600,
      maxLoginAttempts: 5,
      passwordExpiry: 90,
      twoFactor: false
    }
  });

  const [currentTab, setCurrentTab] = useState(0);
  const [hasChanges, setHasChanges] = useState(false);

  const updateConfig = (section: string, field: string, value: any) => {
    setConfig(prev => ({
      ...prev,
      [section]: {
        ...prev[section],
        [field]: value
      }
    }));
    setHasChanges(true);
  };

  const saveConfiguration = async () => {
    try {
      await fetch('/api/config', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(config)
      });
      setHasChanges(false);
      alert('Configuration saved successfully!');
    } catch (error) {
      alert('Failed to save configuration');
    }
  };

  return (
    <div className="app-config">
      <div className="config-header">
        <h1>Application Configuration</h1>
        {hasChanges && (
          <div className="unsaved-changes">
            <span>You have unsaved changes</span>
            <Button primary onClick={saveConfiguration}>
              Save All Changes
            </Button>
          </div>
        )}
      </div>
      
      <TabStrip
        selected={currentTab}
        onSelect={(e) => setCurrentTab(e.selected)}
        tabPosition="left"
        className="config-tabs"
      >
        <Tab title="General Settings">
          <div className="config-section">
            <h3>General Application Settings</h3>
            
            <div className="config-form">
              <div className="form-group">
                <label>Application Name</label>
                <input
                  type="text"
                  value={config.general.appName}
                  onChange={(e) => updateConfig('general', 'appName', e.target.value)}
                />
              </div>
              
              <div className="form-group">
                <label>Language</label>
                <select
                  value={config.general.language}
                  onChange={(e) => updateConfig('general', 'language', e.target.value)}
                >
                  <option value="en-US">English (US)</option>
                  <option value="en-GB">English (UK)</option>
                  <option value="es-ES">Spanish</option>
                  <option value="fr-FR">French</option>
                  <option value="de-DE">German</option>
                </select>
              </div>
              
              <div className="form-group">
                <label>Timezone</label>
                <select
                  value={config.general.timezone}
                  onChange={(e) => updateConfig('general', 'timezone', e.target.value)}

                  <option value="UTC-8">Pacific Time (UTC-8)</option>
                  <option value="UTC-5">Eastern Time (UTC-5)</option>
                  <option value="UTC+0">Greenwich Time (UTC+0)</option>
                  <option value="UTC+1">Central European Time (UTC+1)</option>
                </select>
              </div>
              
              <div className="form-group">
                <label>Theme</label>
                <select
                  value={config.general.theme}
                  onChange={(e) => updateConfig('general', 'theme', e.target.value)}
                >
                  <option value="light">Light Theme</option>
                  <option value="dark">Dark Theme</option>
                  <option value="auto">Auto (System)</option>
                </select>
              </div>
            </div>
          </div>
        </Tab>
        
        <Tab title="Database">
          <div className="config-section">
            <h3>Database Configuration</h3>
            
            <div className="config-form">
              <div className="form-group">
                <label>Database Host</label>
                <input
                  type="text"
                  value={config.database.host}

                />
              </div>
              
              <div className="form-group">

                />
              </div>

                />

                    onChange={(e) => updateConfig('database', 'ssl', e.target.checked)}
                  />
                  Enable SSL Connection
                </label>
              </div>
              
              <div className="form-actions">
                <Button>Test Connection</Button>
              </div>
            </div>
          </div>
```
